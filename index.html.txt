    <script>
        const noBtn = document.getElementById('no-btn');
        const yesBtn = document.getElementById('yes-btn');
        const quizBox = document.getElementById('quiz-box');
        const resultBox = document.getElementById('result-box');

        // دالة تحريك زر "لا" إلى مكان عشوائي في الشاشة
        function moveNoButton() {
            // حساب أبعاد الشاشة المتاحة للحركة مع ترك مسافة أمان
            const padding = 50;
            const maxX = window.innerWidth - noBtn.offsetWidth - padding;
            const maxY = window.innerHeight - noBtn.offsetHeight - padding;

            // توليد إحداثيات عشوائية تماماً
            const randomX = Math.max(padding, Math.floor(Math.random() * maxX));
            const randomY = Math.max(padding, Math.floor(Math.random() * maxY));

            // تحويل طريقة العرض لتتحرك بحرية في كل الشاشة
            noBtn.style.position = 'fixed';
            noBtn.style.left = randomX + 'px';
            noBtn.style.top = randomY + 'px';
        }

        // هروب الزر عند اقتراب الماوس (للكمبيوتر) وعند اللمس (للموبايل)
        noBtn.addEventListener('mouseenter', moveNoButton);
        noBtn.addEventListener('touchstart', function(e) {
            e.preventDefault(); // منع النقرة الأصلية
            moveNoButton();
        });
        noBtn.addEventListener('click', moveNoButton);

        // دالة إظهار كلمة "جدعة" عند الضغط على أيوة
        function showSuccess() {
            quizBox.style.display = 'none';
            resultBox.style.display = 'block';
        }
    </script>
